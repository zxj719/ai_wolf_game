```javascript
// R108: 摄梦人 DAY_SPEECH 平安夜双来源推断三层体系（单夜→两连→三连）
const isPeacefulNightDW = ctx.dayCount > 1 && ctx.lastNightInfo?.includes('平安夜');
const dwPrevDay = ctx.dayCount > 1 ? ctx.dayCount - 1 : 0;
const isConsecutivePeacefulDW = ctx.dayCount >= 3 && isPeacefulNightDW &&
    ctx.fullGameTimeline?.includes(`N${ctx.dayCount - 2}:平安夜`);
const dwPrevPrevDay = ctx.dayCount >= 3 ? ctx.dayCount - 2 : 0;
const dreamTargetRef = lastDreamTarget !== null ? `${lastDreamTarget}号` : '（记录缺失）';

let dwPeaceNightStep = '';
if (isPeacefulNightDW) {
    const isTripleConsecutivePeacefulDW = ctx.dayCount >= 4 && isConsecutivePeacefulDW &&
        ctx.fullGameTimeline?.includes(`N${ctx.dayCount - 3}:平安夜`);
    const dwThreePrevDay = ctx.dayCount >= 4 ? ctx.dayCount - 3 : 0;

    const tripleConsecutivePeaceHintDW = isTripleConsecutivePeacefulDW ? `...三阶...` : '';
    const consecutivePeaceHintDW = isConsecutivePeacefulDW ? `${tripleConsecutivePeaceHintDW}⭕【两连...】` : '';
    dwPeaceNightStep = `${consecutivePeaceHintDW}⭕【摄梦人平安夜双来源推断...】`;
}
```

### 摄梦人 DAY_SPEECH 双来源推断独特性
- **来源A**：狼人攻击入梦目标 → 入梦免疫拦截（入梦目标 = 狼刀目标；confidence ≥65 时来源A更可信；今晚维持入梦该目标）
- **来源B**：守卫/女巫保护他处 → 入梦目标非狼刀目标（confidence <65 时来源B更可信；考虑今晚调整入梦方向）
- **vs 骑士 DAY_SPEECH**（R101）：骑士推断结论是"决斗候选 confidence 下降"；摄梦人推断结论是"入梦目标 confidence 上升 + 今晚是否维持入梦"——**角色决策框架不同，推断结论不能套用同一模板**

## Git 状态恢复记录（R108-A）

本轮启动时发现：
- 状态：detached HEAD（SHA 614c40e，与 origin/main 相同）
- 问题：`git checkout main` 后显示 "local main 落后 remote 50 commits"（local main 停在 fa236cb = R86 时代，remote 已在 R123）
- **原因**：远端多轮并行工作后 force push 了 main，导致 local main 引用指向旧 SHA
- **解决**：`git reset --hard origin/main`（工作树干净，无丢失）
- **预防**：每次工作开始先 `git fetch origin main && git log --oneline HEAD..origin/main` 检查是否落后
